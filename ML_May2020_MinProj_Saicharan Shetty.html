def feature(x): 
    iqr = eda[x].quantile(0.75)-eda[x].quantile(0.25) 
    ulimit= eda[x].quantile(0.75) + 1.5*iqr 
    llimit = eda[x].quantile(0.25) - 1.5*iqr 
    outlier = eda[(eda[x]>=ulimit) | (eda[x]<=llimit)]
    c=outlier[x].count()
    y=pd.DataFrame([iqr,ulimit,llimit,c],index=['iqr','ulimit','llimit','outliercount'])
    return y
def handle(x):
    iqr = eda[x].quantile(0.75)-eda[x].quantile(0.25) 
    ulimit= eda[x].quantile(0.75) + 1.5*iqr 
    llimit = eda[x].quantile(0.25) - 1.5*iqr 
    eda.loc[eda[x]>=ulimit,x]=eda[x].quantile(0.95)
    eda.loc[eda[x]<=llimit,x]=eda[x].quantile(0.05)
    return 0
